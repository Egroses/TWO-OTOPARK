void gotoxy(short x,short y)
{
    HANDLE hConsoleOutput;
    COORD Cursor_Pos={x-1,y-1};
    hConsoleOutput=GetStdHandle(STD_OUTPUT_HANDLE);
    SetConsoleCursorPosition(hConsoleOutput,Cursor_Pos);
}
